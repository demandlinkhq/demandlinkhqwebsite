<script>
(function () {
  'use strict';

  /* ── Close mobile menu ────────────────────────────────── */
  function closeMobileMenu() {
    var header = document.getElementById('dl-header');
    var btn    = document.getElementById('dl-burger-btn');
    if (header) header.classList.remove('menu-open');
    if (btn)    btn.setAttribute('aria-expanded', 'false');
  }

  /* ── Toggle mobile menu ───────────────────────────────── */
  window.dlToggleMenu = function () {
    var header = document.getElementById('dl-header');
    var btn    = document.getElementById('dl-burger-btn');
    if (!header) return;
    var isOpen = header.classList.toggle('menu-open');
    if (btn) btn.setAttribute('aria-expanded', String(isOpen));
  };

  /* ── Smooth scroll with header-height offset ──────────── */
  window.dlScroll = function (id) {
    var target = document.getElementById(id);
    if (!target) return;
    var header   = document.getElementById('dl-header');
    var isStuck  = header && !header.classList.contains('is-unstuck');
    var offset   = isStuck ? header.offsetHeight : 0;
    var top      = target.getBoundingClientRect().top + window.pageYOffset - offset;
    window.scrollTo({ top: top, behavior: 'smooth' });
    closeMobileMenu();
  };

  /* ── Un-sticky when contact section reaches header bottom ─
     Switches position from sticky to relative so the contact
     section fills the full viewport top without the header
     overlapping it. Restores sticky on scroll back up.      */
  function initStickyGuard() {
    var header  = document.getElementById('dl-header');
    var contact = document.getElementById('dl-contact');

    if (!header || !contact) {
      /* Other blocks may not have rendered yet — retry */
      setTimeout(initStickyGuard, 300);
      return;
    }

    function tick() {
      var contactTop = contact.getBoundingClientRect().top;
      if (contactTop <= header.offsetHeight) {
        header.classList.add('is-unstuck');
      } else {
        header.classList.remove('is-unstuck');
      }
    }

    window.addEventListener('scroll', tick, { passive: true });
    tick();
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', initStickyGuard);
  } else {
    initStickyGuard();
  }

}());
</script>
